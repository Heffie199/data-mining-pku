\begin{algorithm}
	\caption{Q-learning迭代算法}
	\label{algorithm-Q-learning}
	\begin{algorithmic}
		\STATE 随机初始化$ Q[num\_states, num\_actions] $
		\STATE 获得初始状态s
		\REPEAT
		\STATE 选择并执行一个动作a
		\STATE 获得奖励r和新状态$ s^{'} $
		\STATE $ Q[s, a] = Q[s, a] + \alpha(r +　\gamma \max_{a^{'}}Q[s^{'}, a^{'}] - Q[s, a]) $
		\STATE $ s = s^{'} $
		\UNTIL 终止		
	\end{algorithmic}		
\end{algorithm}
